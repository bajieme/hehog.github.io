# 一、定义抽象类

# 计算机产品抽象类
import org.apache.commons.lang3.builder.ReflectionToStringBuilder;
import org.apache.commons.lang3.builder.ToStringStyle;

public abstract class AbstractComputer {

    /**
     * cpu
     */
    protected String cpu;

    /**
     * 内存
     */
    protected String memory;

    /**
     * 硬盘
     */
    protected String disk;

    /**
     * 系统
     */
    protected String os;

    protected AbstractComputer() {
    }

    public void setCpu(String cpu) {
        this.cpu = cpu;
    }

    public void setDisk(String disk) {
        this.disk = disk;
    }

    public void setMemory(String memory) {
        this.memory = memory;
    }

    /**
     * 安装系统
     */
    public abstract void setOs();

    @Override
    public String toString() {
        return ReflectionToStringBuilder.toString(this, ToStringStyle.SHORT_PREFIX_STYLE);
    }
}

    @Override
    public void createCpu(String cpu) {

    public void createMemory(String memory) {

    }

    @Override
    public void createOs() {
        computer.setOs();
    }

    @Override
    public AbstractComputer build() {
        return computer;
    }

# 五、ComputerDirector类，负责具体的构造产品Computer

public class ComputerDirector {
    AbstractComputerBuilder builder;


    public ComputerDirector(AbstractComputerBuilder builder) {
        this.builder = builder;
    }

    public void setBuilder(AbstractComputerBuilder builder) {
        this.builder = builder;
    }

    public void construct(String cpu,String memory,String disk){
        builder.createCpu(cpu);
        builder.createMemory(memory);
        builder.createDisk(disk);
        builder.createOs();
    }

    public void construct(){
        builder.createOs();
    }
}

# 六、具体使用

public class MainBuilderTest {

    public static void main(String[] args) {
        AbstractComputerBuilder macBuilder = new MacBookBuilder();
        ComputerDirector director = new ComputerDirector(macBuilder);
        director.construct("i5", "16g", "256g固态");
        AbstractComputer macBook = macBuilder.build();
        System.out.println("得到苹果 MacBook 配置: " + macBook);

        MateBookBuilder mateBookBuilder = new MateBookBuilder();
        director.setBuilder(mateBookBuilder);
        director.construct("i7", "32g", "1T固态");
        AbstractComputer mateBook = mateBookBuilder.build();
        System.out.println("得到华为 MateBook 配置: " + mateBook);

    }
}
